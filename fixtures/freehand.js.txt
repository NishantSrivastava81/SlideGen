function addHeader(slide, title, subtitle) {
  slide.addShape(pptx.ShapeType.rect, {
    x: 0, y: 0, w: 13.333, h: 0.65,
    line: { color: C.navy, transparency: 100 },
    fill: { color: C.navy }
  });
  slide.addText(title, {
    x: 0.4, y: 0.14, w: 7.2, h: 0.28,
    fontFace: 'Aptos Display',
    fontSize: 24,
    bold: true,
    color: C.white,
    margin: 0
  });
  if (subtitle) {
    slide.addText(subtitle, {
      x: 8.0, y: 0.17, w: 4.9, h: 0.22,
      align: 'right',
      fontSize: 10,
      color: 'D6E4FF',
      margin: 0
    });
  }
}

function addFooter(slide, text) {
  slide.addLine({
    x: 0.35, y: 7.12, w: 12.6, h: 0,
    line: { color: C.gray300, pt: 1 }

  slide.addText(text, {
    x: 0.4, y: 7.18, w: 6.5, h: 0.18,
    fontSize: 8,
    color: C.gray500,
    margin: 0
  });
}

function addKpiCard(slide, x, y, w, h, accent, title, value, body) {
  slide.addShape(pptx.ShapeType.roundRect, {
    x, y, w, h,
    rectRadius: 0.06,
    line: { color: C.gray200, pt: 1 },
    fill: { color: C.white },

  });
  slide.addShape(pptx.ShapeType.rect, {
    x, y, w: 0.12, h,
    line: { color: accent, transparency: 100 },
    fill: { color: accent }
  });
  slide.addText(title, {
    x: x + 0.22, y: y + 0.18, w: w - 0.35, h: 0.22,
    fontSize: 11, bold: true, color: C.gray700, margin: 0
  });
  slide.addText(value, {
    x: x + 0.22, y: y + 0.52, w: w - 0.35, h: 0.38,
    fontSize: 21, bold: true, color: accent, margin: 0
  });
  slide.addText(body, {
    x: x + 0.22, y: y + 1.0, w: w - 0.35, h: h - 1.15,
    fontSize: 9.5, color: C.gray700, valign: 'top', breakLine: false,
    margin: 0.02
  });
}

function addBulletList(slide, x, y, w, items, color = C.gray900, fontSize = 13) {
  const runs = [];
  items.forEach((t) => {
    runs.push({ text: t, options: { bullet: { indent: 12 } } });
  });
  slide.addText(runs, {
    x, y, w, h: 2.2,
    fontSize,
    color,
    breakLine: true,
    paraSpaceAfterPt: 10,
    valign: 'top',
    margin: 0.02
  });
}

function addSectionLabel(slide, x, y, text, fill) {
  slide.addShape(pptx.ShapeType.roundRect, {
    x, y, w: 1.55, h: 0.34,
    rectRadius: 0.05,
    line: { color: fill, transparency: 100 },
    fill: { color: fill }
  });

    x: x + 0.08, y: y + 0.07, w: 1.39, h: 0.16,
    fontSize: 9, bold: true, color: C.white, align: 'center', margin: 0
  });
}
